    <script>
        let cart = [];
        let orderCount = 0;
        let selectedPayment = 'phonepe';
        
        function updateQuantity(itemName, change, price) {
            const inputElement = document.getElementById(${itemName}-qty);
            let quantity = parseInt(inputElement.value);
            
            quantity += change;
            if (quantity < 0) quantity = 0;
            
            inputElement.value = quantity;
            
            // Update cart
            const existingItemIndex = cart.findIndex(item => item.name === itemName);
            
            if (existingItemIndex !== -1) {
                if (quantity === 0) {
                    // Remove item from cart if quantity is 0
                    cart.splice(existingItemIndex, 1);
                } else {
                    // Update quantity
                    cart[existingItemIndex].quantity = quantity;
                }
            } else if (quantity > 0) {
                // Add new item to cart
                cart.push({
                    name: itemName,
                    price: price,
                    quantity: quantity
                });
            }
            
            updateCart();
        }
        
        function updateCart() {
            const cartItems = document.getElementById('cart-items');
            const cartTotal = document.getElementById('cart-total');
            
            if (cart.length === 0) {
                cartItems.innerHTML = '<p>Your cart is empty. Please add some items from the menu.</p>';
                cartTotal.textContent = 'Total: ₹0';
                return;
            }
            
            let cartHTML = '';
            let total = 0;
            
            cart.forEach(item => {
                const itemTotal = item.price * item.quantity;
                total += itemTotal;
                
                cartHTML += `
                    <div class="cart-item">
                        <div>${item.name} x ${item.quantity}</div>
                        <div>₹${itemTotal}</div>
                    </div>
                `;
            });
            
            cartItems.innerHTML = cartHTML;
            cartTotal.textContent = Total: ₹${total};
        }
        
        function proceedToCheckout() {
            if (cart.length === 0) {
                alert('Please add at least one item to your cart before proceeding to checkout.');
                return;
            }
            
            document.getElementById('checkout-form').style.display = 'block';
            document.getElementById('checkout-form').scrollIntoView({ behavior: 'smooth' });
        }
        
        function selectPayment(method) {
            selectedPayment = method;
            
            // Update UI
            document.querySelectorAll('.payment-option').forEach(option => {
                option.classList.remove('selected');
            });
            
            document.querySelector(.payment-option:${method === 'phonepe' ? 'first-child' : 'last-child'}).classList.add('selected');
        }
        
        function processPayment() {
            const name = document.getElementById('name').value;
            const phone = document.getElementById('phone').value;
            const address = document.getElementById('address').value;
            
            if (!name || !phone || !address) {
                alert('Please fill in all required fields.');
                return;
            }
            
            // Calculate total
            let total = 0;
            cart.forEach(item => {
                total += item.price * item.quantity;
            });
            
            if (selectedPayment === 'phonepe') {
                // Redirect to PhonePe payment
                const phonePeURL = phonepe://pay?pa=9958965893@ybl&pn=Spice%20Delight&am=${total}&cu=INR;
                window.location.href = phonePeURL;
                
                // For demo purposes, we'll show the success modal after a delay
                setTimeout(() => {
                    showSuccessModal();
                    sendWhatsAppOrder(name, phone, address, total);
                }, 3000);
            } else {
                // Cash on delivery
                showSuccessModal();
                sendWhatsAppOrder(name, phone, address, total);
            }
        }
        
        function showSuccessModal() {
            orderCount++;
            document.getElementById('order-id').textContent = SD${1000 + orderCount};
            document.getElementById('success-modal').style.display = 'flex';
        }
        
        function closeModal() {
            document.getElementById('success-modal').style.display = 'none';
            
            // Reset form
            cart = [];
            updateCart();
            document.getElementById('checkout-form').style.display = 'none';
            document.getElementById('name').value = '';
            document.getElementById('phone').value = '';
            document.getElementById('address').value = '';
            
            // Reset all quantity inputs
            document.querySelectorAll('.quantity-input').forEach(input => {
                input.value = '0';
            });
        }

            orderDetails += Payment Method: ${selectedPayment === 'phonepe' ? 'PhonePe' : 'Cash on Delivery'};
            
            // Create WhatsApp URL
            const whatsappURL = https://wa.me/917834905488?text=${orderDetails};
